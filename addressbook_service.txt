create database addressbook_service

// switch to a database

//creating a table inside of database
create table addressbook{
FirstName varchar(50),
LastName varchar(50),
Address  varchar(20),
City     varchar(20),
State    varchar(20),
Zip    smallint,
PhoneNumber bigint,
Email   varchar(40)
);
